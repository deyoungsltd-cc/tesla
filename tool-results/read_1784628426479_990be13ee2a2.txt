     6→// ─── Intersection Observer Hook ─────────────────────────────────────────────
     7→function useInView(threshold = 0.15) {
     8→  const ref = useRef<HTMLDivElement>(null);
     9→  const [isVisible, setIsVisible] = useState(false);
    10→
    11→  useEffect(() => {
    12→    const el = ref.current;
    13→    if (!el) return;
    14→    const observer = new IntersectionObserver(
    15→      ([entry]) => {
    16→        if (entry.isIntersecting) {
    17→          setIsVisible(true);
    18→          observer.unobserve(el);
    19→        }
    20→      },
    21→      { threshold }
    22→    );
    23→    observer.observe(el);
    24→    return () => observer.disconnect();
    25→  }, [threshold]);
    26→
    27→  return { ref, isVisible };
    28→}

    30→function FadeIn({
    31→  children,
    32→  className = '',
    33→  delay = 0,
    34→  direction = 'up',
    35→}: {
    36→  children: ReactNode;
    37→  className?: string;
    38→  delay?: number;
    39→  direction?: 'up' | 'down' | 'left' | 'right';
    40→}) {
    41→  const { ref, isVisible } = useInView(0.1);
    42→  const transforms: Record<string, string> = {
    43→    up: 'translateY(32px)',
    44→    down: 'translateY(-32px)',
    45→    left: 'translateX(32px)',
    46→    right: 'translateX(-32px)',
    47→  };
    48→
    49→  return (
    50→    <div
    51→      ref={ref}
    52→      className={className}
    53→      style={{
    54→        opacity: isVisible ? 1 : 0,
    55→        transform: isVisible ? 'translateY(0) translateX(0)' : transforms[direction],
    56→        transition: `opacity 0.7s cubic-bezier(0.16, 1, 0.3, 1) ${delay}s, transform 0.7s cubic-bezier(0.16, 1, 0.3, 1) ${delay}s`,
    57→      }}
    58→    >
    59→      {children}
    60→    </div>
    61→  );
    62→}
