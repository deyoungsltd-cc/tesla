     5→// ─── Intersection Observer Hook ─────────────────────────────────────────────
     6→function useInView(threshold = 0.15) {
     7→  const ref = useRef<HTMLDivElement>(null);
     8→  const [isVisible, setIsVisible] = useState(false);
     9→
    10→  useEffect(() => {
    11→    const el = ref.current;
    12→    if (!el) return;
    13→    const observer = new IntersectionObserver(
    14→      ([entry]) => {
    15→        if (entry.isIntersecting) {
    16→          setIsVisible(true);
    17→          observer.unobserve(el);
    18→        }
    19→      },
    20→      { threshold }
    21→    );
    22→    observer.observe(el);
    23→    return () => observer.disconnect();
    24→  }, [threshold]);
    25→
    26→  return { ref, isVisible };
    27→}

    29→function FadeIn({
    30→  children,
    31→  className = '',
    32→  delay = 0,
    33→  direction = 'up',
    34→}: {
    35→  children: ReactNode;
    36→  className?: string;
    37→  delay?: number;
    38→  direction?: 'up' | 'down' | 'left' | 'right';
    39→}) {
    40→  const { ref, isVisible } = useInView(0.1);
    41→  const transforms: Record<string, string> = {
    42→    up: 'translateY(40px)',
    43→    down: 'translateY(-40px)',
    44→    left: 'translateX(40px)',
    45→    right: 'translateX(-40px)',
    46→  };
    47→
    48→  return (
    49→    <div
    50→      ref={ref}
    51→      className={className}
    52→      style={{
    53→        opacity: isVisible ? 1 : 0,
    54→        transform: isVisible ? 'translateY(0) translateX(0)' : transforms[direction],
    55→        transition: `opacity 0.8s cubic-bezier(0.16, 1, 0.3, 1) ${delay}s, transform 0.8s cubic-bezier(0.16, 1, 0.3, 1) ${delay}s`,
    56→      }}
    57→    >
    58→      {children}
    59→    </div>
    60→  );
    61→}
